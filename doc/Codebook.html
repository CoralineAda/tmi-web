  {
    explainer: "// Show me a graph of \"#{context}\"-related categories and their associated codes.",
    query: "MATCH (cat:Category)-[:CATEGORIZED_AS]-(c:Code) WHERE cat.context=\"#{context}\" RETURN cat,c"
